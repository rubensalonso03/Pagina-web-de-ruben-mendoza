        document.addEventListener('DOMContentLoaded', function() {
            // =======================================================
            // LÓGICA DE INDICADOR DE SCROLL (NUEVO)
            // =======================================================
            const scrollProgress = document.getElementById('scroll-progress');
            
            function updateScrollProgress() {
                const totalHeight = document.body.scrollHeight - window.innerHeight;
                const scrollPosition = window.scrollY;
                const progress = (scrollPosition / totalHeight) * 100;
                
                // Asegurar que el ancho es 0 si estamos al inicio y 100 si estamos al final
                scrollProgress.style.width = `${Math.min(100, Math.max(0, progress))}%`;
            }

            // Añadir listener para actualizar el progreso al hacer scroll
            window.addEventListener('scroll', updateScrollProgress);
            // Llamar al inicio para establecer la posición inicial
            updateScrollProgress();

            // =======================================================
            // LÓGICA DE DESPLAZAMIENTO SUAVE
            // =======================================================
            // Seleccionar todos los enlaces de navegación que contienen un hash (#)
            const navLinks = document.querySelectorAll('#main-nav a[href^="#"]');

            navLinks.forEach(link => {
                link.addEventListener('click', function(e) {
                    e.preventDefault(); // Detener el comportamiento de salto predeterminado

                    const targetId = this.getAttribute('href'); // Obtener el ID del destino (ej: #sobre-mi o #hero)
                    const targetElement = document.querySelector(targetId);
                    
                    if (targetElement) {
                        // Calcular la posición para compensar el header fijo
                        const headerHeight = document.querySelector('header').offsetHeight;
                        const targetPosition = targetElement.offsetTop - headerHeight - 10; // 10px de margen

                        window.scrollTo({
                            top: targetPosition,
                            behavior: 'smooth'
                        });
                    }
                });
            });


            // =======================================================
            // LÓGICA DE MODO OSCURO/CLARO
            // =======================================================
            const themeToggle = document.getElementById('theme-toggle');
            const body = document.body;
            const moonIcon = document.getElementById('moon-icon');
            const sunIcon = document.getElementById('sun-icon');

            function cargarTema() {
                const isLight = localStorage.getItem('theme') === 'light';
                body.classList.toggle('light-mode', isLight);
                moonIcon.classList.toggle('hidden', isLight);
                sunIcon.classList.toggle('hidden', !isLight);
            }

            function alternarTema() {
                const isLight = body.classList.toggle('light-mode');
                localStorage.setItem('theme', isLight ? 'light' : 'dark');
                moonIcon.classList.toggle('hidden', isLight);
                sunIcon.classList.toggle('hidden', !isLight);
            }

            cargarTema();
            if (themeToggle) {
                themeToggle.addEventListener('click', alternarTema);
            }

            // =======================================================
            // LÓGICA DE MODAL
            // =======================================================
            const abrirModal = document.getElementById('abrirModal');
            const cerrarModalBtn = document.getElementById('cerrarModal');
            const ventanaModal = document.getElementById('ventanaModal');

            function abrirModalHandler() {
                if (ventanaModal) {
                    ventanaModal.classList.add('abierto');
                }
            }

            function cerrarModal() {
                if (ventanaModal) {
                    ventanaModal.classList.remove('abierto');
                }
            }

            if (abrirModal) {
                abrirModal.addEventListener('click', abrirModalHandler);
            }
            if (cerrarModalBtn) {
                cerrarModalBtn.addEventListener('click', cerrarModal);
            }
            if (ventanaModal) {
                ventanaModal.addEventListener('click', (e) => {
                    if (e.target === ventanaModal) {
                        cerrarModal();
                    }
                });
            }


            // =======================================================
            // LÓGICA DE SUSCRIPCIÓN
            // =======================================================

            const botonSuscribir = document.getElementById('botonSuscribir');
            
            function alternarSuscripcion() {
                // Comprueba si ya está suscrito
                if (botonSuscribir.classList.contains('suscrito')) {
                    // DESUSCRIBIRSE
                    botonSuscribir.textContent = 'Suscribirse al Newsletter';
                    botonSuscribir.classList.remove('suscrito');
                    // Restaurar clases originales
                    botonSuscribir.classList.add('bg-red-600', 'hover:bg-red-700', 'border-red-700');
                    botonSuscribir.classList.remove('bg-gray-700', 'hover:bg-gray-600', 'border-gray-600');
                } else {
                    // SUSCRIBIRSE
                    botonSuscribir.textContent = 'Suscrito ✓';
                    botonSuscribir.classList.add('suscrito');
                    // Eliminar clases originales que entran en conflicto con .suscrito
                    botonSuscribir.classList.remove('bg-red-600', 'hover:bg-red-700', 'border-red-700');
                }
            }

            if (botonSuscribir) {
                botonSuscribir.addEventListener('click', alternarSuscripcion);
            }
            
            // =======================================================
            // LÓGICA DE CALCULADORA DE ESFUERZO
            // =======================================================

            const fuerzaInput = document.getElementById('fuerzaInput');
            const areaInput = document.getElementById('areaInput');
            const resultadoTension = document.getElementById('resultadoTension');
            const calcularBtn = document.getElementById('calcularBtn');

            function calcularTension() {
                const F = parseFloat(fuerzaInput.value);
                const A = parseFloat(areaInput.value);

                // Validation
                if (isNaN(F) || isNaN(A) || A <= 0) {
                    resultadoTension.textContent = 'Error: Ingresa valores válidos.';
                    return;
                }

                // Calculation: Stress (σ) = Force (F) / Area (A)
                const tension = F / A;
                
                // Display result, limiting to 4 decimal places
                resultadoTension.textContent = `${tension.toFixed(4)} Pa`;
            }

            if (calcularBtn) {
                calcularBtn.addEventListener('click', calcularTension);
            }


            // =======================================================
            // LÓGICA DE VALIDACIÓN DE FORMULARIO DE CONTACTO
            // =======================================================

            const MIN_MESSAGE_LENGTH = 10; 

            // Referencias del FORMULARIO DE CONTACTO
            const contactForm = document.getElementById('contact-form');
            const nameInput = document.getElementById('name-input');
            const emailInput = document.getElementById('email-input');
            const messageInput = document.getElementById('message-input'); 
            const contactSuccessMessage = document.getElementById('contact-success-message'); 
            
            // Referencias a los SPAN de error contextual
            const errorNameEmptySpan = document.getElementById('error-name-empty'); 
            const errorEmailFormatSpan = document.getElementById('error-email-format');
            const errorMsgLengthSpan = document.getElementById('error-message-length'); 

            // Variable global para manejar el tiempo de espera de mensajes
            let errorTimeout; 

            /**
             * Oculta todos los spans de error contextual.
             */
            function hideAllContextualErrors() {
                 [errorNameEmptySpan, errorEmailFormatSpan, errorMsgLengthSpan].forEach(span => {
                    span.style.display = 'none';
                    span.textContent = ''; // Limpiar contenido
                 });
            }

            /**
             * Muestra mensajes de éxito globales con un temporizador.
             */
            function showUIMessage(messageElement, duration = 4000) {
                // OCULTAR TODOS LOS SPANS CONTEXTUALES ANTES DE MOSTRAR UN MENSAJE GLOBAL
                hideAllContextualErrors();

                contactSuccessMessage.classList.remove('visible'); 

                clearTimeout(errorTimeout); 

                messageElement.classList.add('visible');

                errorTimeout = setTimeout(() => {
                    messageElement.classList.remove('visible');
                }, duration);
            }

            // --- EVENT LISTENER para el Formulario de Contacto ---
            if (contactForm) {
                contactForm.addEventListener('submit', function(e) {
                    
                    e.preventDefault(); // Interceptar el envío

                    // 1. Ocultar todos los errores contextuales antes de validar
                    hideAllContextualErrors();

                    const nameValue = nameInput.value.trim();
                    const emailValue = emailInput.value.trim();
                    const messageValue = messageInput.value.trim();
                    
                    let isValid = true; // Variable de control

                    // 2. VALIDACIÓN 1: Nombre no vacío
                    if (nameValue === '') {
                        errorNameEmptySpan.textContent = 'El nombre es obligatorio.';
                        errorNameEmptySpan.style.display = 'block';
                        isValid = false;
                    }

                    // 3. VALIDACIÓN 2: Email contiene '@'
                    if (!emailValue.includes('@')) {
                        errorEmailFormatSpan.textContent = 'Debe ser un email válido (contener "@").';
                        errorEmailFormatSpan.style.display = 'block';
                        isValid = false;
                    }

                    // 4. VALIDACIÓN 3: Longitud mínima del mensaje
                    if (messageValue.length < MIN_MESSAGE_LENGTH) {
                        errorMsgLengthSpan.textContent = `El mensaje debe tener al menos ${MIN_MESSAGE_LENGTH} caracteres.`;
                        errorMsgLengthSpan.style.display = 'block';
                        isValid = false;
                    }

                    // 5. VALIDACIÓN COMPLETA
                    if (!isValid) {
                        console.warn("Validación fallida. Favor de corregir los campos.");
                        // Si alguna validación falló, detenemos la función
                        return; 
                    }

                    // Si todas las validaciones son exitosas:
                    console.log("¡Formulario enviado con éxito!");
                    console.log("Datos:", { name: nameValue, email: emailValue, message: messageValue });

                    // Mostrar mensaje de éxito global
                    showUIMessage(contactSuccessMessage);

                    // Limpiar formulario (Simular éxito de envío)
                    contactForm.reset(); 
                });
            }

        });
